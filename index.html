function doPost(e) { const params = JSON.parse(e.postData.contents); let result; if (params.action === 'verify') { result = verifyUserCode(params.storeId, params.staffId, params.uid); } else if (params.action === 'register') { result = registerUser(params); } else if (params.action === 'checkAdmin') { result = checkAdmin(params.uid); } else if (params.action === 'saveStore') { result = saveStore(params); } else if (params.action === 'saveSettings') { result = saveSettings(params); } else if (params.action === 'getShiftTemplate') { result = getShiftTemplate(params.storeId); } else if (params.action === 'saveShiftTemplate') { result = saveShiftTemplate(params); } else if (params.action === 'getHolidays') { result = getHolidays(params); } else if (params.action === 'getShiftOverrides') { result = getShiftOverrides(params); } else if (params.action === 'saveShiftOverride') { result = saveShiftOverride(params); } else { result = { success: false, msg: '不明なアクション' }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); }
